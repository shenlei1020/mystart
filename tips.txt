docker tips:
docker run --hostname tensorflow --name tensorflow  -v $PWD\workspace:/root/workspace -it saturnlab/tensorflow:0.3.0 --启动container，挂在文件，设置用户名

docker build --从dockerfile建立image

docker commit --对产生更改的container从新保存为image

docker cp --从container之间、container和

docker images --显示当前已有的镜像

docker rm CI --删除container

docker rmi ImageID --删除

docker ps -a --显示当前的container

docker run -v $PWD/workspace/ubuntu:/root/ubuntu -it -p 3000:3000 ubuntu:xenial --将当前路径映射到虚拟机ubuntu下的/root/ubuntu 

docker system prune --删除stopped的container、最新container不使用的network、挂载container

docker ps -a --显示所有的container进程

docker run -it saturnlab/tensorflow:0.3.0 --从一个image进入一个container

exit --调出container，但是container仍然在run

docker stop CI --停止某个container

docker start CI --开启某个container

docker exec -it CI bash --进入某个container
